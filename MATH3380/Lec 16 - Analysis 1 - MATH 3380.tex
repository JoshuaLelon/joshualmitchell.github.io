\bsc{Limit Superior \& Limit Inferior}{

\ssc{Definition 4.4.9}{

Let \bk{\uw{s}{n}} be a bounded sequence.

A \textbf{subsequential limit} of \bk{\uw{s}{n}} is a real number s such that s \eql \lmti{k} \dbs{s}{n}{k} for some subsequence \bk{\dbs{s}{n}{k}}.

If S \eql \bk{s \mem \br : \lmti{k} \dbs{s}{n}{k} \eql s for some \bk{\dbs{s}{n}{k}} of \bk{\uw{s}{n}}}, then

\balist
\item the \textbf{limit superior} (or \textbf{upper limit}) of \bk{\uw{s}{n}} is given by lim sup \uw{s}{n} \eql sup S
\item the \textbf{limit inferior} (or \textbf{lower limit}) of \bk{\uw{s}{n}} is given by lim inf \uw{s}{n} \eql inf S
\item Clearly, lim inf \uw{s}{n} \lse lim sup \uw{s}{n}. If it happens that lim inf \uw{s}{n} \ls lim sup \uw{s}{n}, then we say that \bk{\uw{s}{n}} \textbf{oscillates}.
\elist

}

\sidenote{
\av{\uw{s}{n}} \lse M, \fa n \mem \bn

$-$M \ls \uw{s}{n} \ls M

If \lmti{k}\dbs{s}{n}{k} \eql s \mem S, then

$-$M \ls \dbs{s}{n}{k} \ls M, so

$-$M \ls s \ls M

\#18, page 179
}

\ssc{Theorem 1}{A bounded sequence \bk{\uw{s}{n}} converges to s iff lim inf \uw{s}{n} \eql lim sup \uw{s}{n} \eql s}

\bgpf
\lra 
Assume \bk{\uw{s}{n}} converges to s.

By Theorem 4.4.4, S \eql \bk{s} (contains only one element).

Then, 

lim inf \uw{s}{n} \eql inf S \eql s

lim sup \uw{s}{n} \eql sup S \eql s

So,

lim inf \uw{s}{n} \eql lim sup \uw{s}{n} \eql s

\lla 

(see HW 8, Exercise 9, page 194)

\epf
}

\ssc{Example 4.4.10}{

\lt{\uw{s}{n} \eql \prn{$-$1}$^n$ \ps \frc{1}{n}}

Show that

lim inf \uw{s}{n} \eql $-$1,

lim sup \uw{s}{n} \eql 1

Notice that if

n is even \rar \uw{s}{n} \eql 1 \ps \frc{1}{n}

n is odd \rar \uw{s}{n} \eql $-$1 \ps \frc{1}{n}

Thus,

\lmti{k} \uw{s}{2k} \eql 1

\lmti{k} \uw{s}{2k + 1} \eql $-$1

Hence,

lim sup \uw{s}{n} \eql 1

lim inf \uw{s}{n} \eql $-$1

}

\ssc{Corollary 4.4.12}{

Let \bk{\uw{s}{n}} be a bounded sequence and let \uf{s}{*} \eql lim sup \uw{s}{n}, \uw{s}{*} \eql lim inf \uw{s}{n}.

\bgpf

For \ep \eql 1, by Theorem 4.4.11 \bpth{a}, \exs \uw{N}{1} \mem \bn st
\step{s_n < s^* + 1 \textrm{, for n \gre \uw{N}{1}}}{1}
By Theorem 4.4.11, \bpth{b}, for \ep \eql 1, i \eql \uw{N}{1}, \exs \uw{n}{1} \gr i \eql \uw{N}{1} st
\eqn{\dbs{s}{n}{1} > s^* - 1}
and
\eqn{s^* - 1 < \dbs{s}{n}{1} < s^* + 1 \textrm{ using (1)}}
For \ep \eql \frc{1}{2}, \exs \uw{N}{2} \mem \bn st
\step{s_n < s^* + \frac{1}{2} \textrm{ for n \gre \uw{N}{2} using (a)}}{2}
Also, for i \eql max\bk{\uw{n}{1}, \uw{N}{2}}, \exs j \eql \uw{n}{2} \gr i (i.e. \uw{n}{2} \gr \uw{n}{1} and \uw{n}{2} \gr \uw{N}{2}) st
\eqn{s^* - \frac{1}{2} < \dbs{s}{n}{2} < s^* + \frac{1}{2}}

\eqn{s^* - \frac{1}{k} < \dbs{s}{n}{k} < s^* + \frac{1}{k}}

\ssc{Theorem 4.4.14}{
Assume that \bk{\uw{r}{n}} converges to r \mem \br where r \gr 0 and \bk{\uw{s}{n}} is bounded.

Then lim sup \uw{r}{n}\uw{s}{n} \eql r lim sup \uw{s}{n}

\bgpf

\exs \uw{M}{1}, \uw{M}{2} \mem \br st
\eqn{|s_n| \lse M_1 \textrm{ and } |r_n| \lse  M_2 \textrm{, }\fa n \mem \bn}
So,
\eqn{|r_ns_n| \lse M_1M_2}
Thus, the sequence \bk{\uw{r}{n}\uw{s}{n}} is bounded, which means lim sup \uw{r}{n}\uw{s}{n} exists.

By Corollary 4.4.12, \exs a subsequence \bk{\dbs{s}{n}{k}} of \bk{\uw{s}{n}} st \lmti{k} \dbs{s}{n}{k} \eql lim sup \uw{s}{n}

Then, \lmti{k} \dbs{r}{n}{k}\dbs{s}{n}{k} \eql (\lmti{k} \dbs{r}{n}{k})(\lmti{k} \dbs{s}{n}{k}) \eql r lim sup \uw{s}{n}

(i.e. r lim sup \uw{s}{n} is a subsequential limit point of the sequence \bk{\uw{r}{n}\uw{s}{n}})

Thus, r lim sup \uw{s}{n} \lse lim sup \uw{r}{n}\uw{s}{n} \bpth{1}

\

Also, assume that \bk{\dbs{r}{n}{l}\dbs{s}{n}{l}} is a subsequence of \bk{\uw{r}{n}\uw{s}{n}} st
\eqn{\lmti{l} \dbs{r}{n}{l} \dbs{s}{n}{l} = t}
Then,
\eqn{\lmti{l} \dbs{s}{n}{l} = \lmti{l} \dbs{s}{n}{l} ( \lmti{l} \frac{\dbs{r}{n}{l}}{r}) = \lmti{l} \frac{\dbs{s}{n}{l} \dbs{r}{n}{l}}{r} = \frac{t}{r} \lse \textrm{lim sup \uw{s}{n}}}
So, t \lse r lim sup \uw{s}{n} (which says r lim sup \uw{s}{n} is an upper bound)

So, lim sup \uw{r}{n}\uw{s}{n} \lse r lim sup \uw{s}{n} \bpth{2}

\bpth{1} and \bpth{2} imply that lim sup \uw{r}{n} \uw{s}{n} \eql r lim sup \uw{s}{n}
\epf
}

	and S \eql \bk{$-$\infy}
