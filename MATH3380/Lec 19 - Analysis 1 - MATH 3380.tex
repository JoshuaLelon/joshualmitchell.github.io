\ssc{5.1.1: Definition (Limits of Functions)}{

\lt{D \sbs \br, f : D \lra \br, and c \mem D\pr (i.e. c is an accumulation point)}

We say that L \mem \br is a \textbf{limit} of f at c if, \fa \ep \gr 0, \exs $\delta(\epsilon)$ \gr 0 st
\eqn{|f(x) - L| - \epsilon}
wherever x \mem D and 0 \ls \av{x - c} \ls $\delta$

(i.e. the limit as x goes to c of f(x) \eql L)

\eqn{x \pm c}
\eqn{-\delta < x - c < \delta}
\eqn{c - \delta < x < c + \delta}

Recall the definition of a limit:

\eqn{f(x) = \lmti{h} \frac{f(x + h) - f(x)}{h}}

where x is fixed.
}
\ssc{Theorem 5.1.2}{
\lt{f : D \lra \br, c \mem D\pr}

f(U $\cap$ D) \sbs V.

\bgpf

(i.e. \exs $\delta(\epsilon)$ \gr 0 st N*(c, $\delta$) \eql U)

st f(U $\cap$ D) \sbs V

\lla 

The converse is similar.

\epf

\eqn{|f(x) - k| = |k - k| = 0 < \epsilon}

\eqn{|f(x) - c| = |x - c| < \epsilon}
