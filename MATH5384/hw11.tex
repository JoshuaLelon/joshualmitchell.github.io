\bsc{Chapter 10}{
\ssc{Exercise 14}{

Explain why the correspondence x \lra 3x from \uw{Z}{12} to \uw{Z}{10} is not a homomorphism.

}

\ssc{Exercise 15}{

Suppose that $\phi$ is a homomorphism from \uw{Z}{30} to \uw{Z}{30} and Ker $\phi$ \eql \bk{0, 10, 20}. If $\phi$(23) \eql 9, determine all elements that map to 9.

}

\ssc{Exercise 16}{

Prove that there is no homomorphism from \uw{Z}{8} \edp \uw{Z}{2} onto \uw{Z}{4} \edp \uw{Z}{4}.

}

\ssc{Exercise 17}{

Prove that there is no homomorphism from \uw{Z}{16} \edp \uw{Z}{2} onto \uw{Z}{4} \edp \uw{Z}{4}.

}

\ssc{Exercise 18}{

Can there be a homomorphism from \uw{Z}{4} \edp \uw{Z}{4} onto \uw{Z}{8}? Can there be a homomorphism from \uw{Z}{16} onto \uw{Z}{2} \edp \uw{Z}{2}? Explain your answers.

}

\ssc{Exercise 20}{

How many homomorphisms are there from \uw{Z}{20} onto \uw{Z}{8}? How many are there to \uw{Z}{8}?

}

\ssc{Exercise 24}{

Suppose that $\phi$: \uw{Z}{50} \lra \uw{Z}{15} is a group homomorphism with $\phi$(7) \eql 6.

\balist
\item Determine $\phi$(x).
\item Determine the image of $\phi$.
\item Determine the kernel of $\phi$.
\item Determine $\uf{\phi}{-1}$(3). That is, determine the set of all elements that map to 3.
\elist

}
